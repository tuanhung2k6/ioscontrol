# Auto-farm game coins
import random

for i in range(100):
    tap(195, 400)
    sleep(random.uniform(0.2, 0.5))

    color = get_color(195, 400)
    if color == 0xFF0000:
        tap(195, 600)    # Collect reward
        sleep(1)

    if i % 10 == 0:
        screenshot(f"progress_{i}.png")

log("✅ Farming complete!")
auto_farm.lua
-- Auto-farm game coins

for i = 0, 99 do
    tap(195, 400)
    sleep(0.2 + math.random() * 0.3)

    local color = getColor(195, 400)
    if color == 0xFF0000 then
        tap(195, 600)    -- Collect reward
        sleep(1)
    end

    if i % 10 == 0 then
        screenshot("progress_" .. i .. ".png")
    end
end

log("✅ Farming complete!")
